

int fact(int n) {
    if (n <= 1)
        return 1;

    int arg = n - 1;
    int result = fact(arg) * n;
    return result;
}

int fib(int ind, int x) {
    if (ind <= 2)
        return 1;

    int arg1 = ind - 1;
    int arg2 = ind - 2;

    return fib(arg1, x) + fib(arg2, x);
}

    int a, b;
    input(a, b);
    int res = a + b;
    output(res);

    return 0;
}
